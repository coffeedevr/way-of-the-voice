zzz_thuum_clavicus.1001 = {
    type = character_event
	title = "Death By Your Thu'um"
    desc = zzz_thuum_clavicus.1001.desc
    theme = dread

    override_background = {
		reference = burning_building
	}

    left_portrait = {
		character = root
		animation = fear
	}

    right_portrait = {
		character = scope:child
		animation = fear
	}

    lower_left_portrait = {
		character = scope:courtier
	}

    lower_center_portrait = {
		character = scope:vassal
	}

	lower_right_portrait = {
		character = scope:councillor
	}

    trigger = {
        has_character_flag = clavicus_strong_event
        NOT = {
            has_character_flag = clavicus_done
        }
    }

    immediate = {
        random_courtier = {
            limit = {
                NOT = {
                    is_child_of = root
                }
            }
            save_scope_as = courtier
        }
        random_vassal = {
            limit = {
                NOT = {
                    is_councillor_of  = root
                }
            }
            save_scope_as = vassal
        }
        random_child = {
            limit = {
                is_child_of = root
            }
            save_scope_as = child
        }
        random_councillor = {
            limit = {
                NOT = {
                    is_child_of = root
                }
            }
            save_scope_as = councillor
        }
    }

    option = {
        name = "Oh no no no no no!!!!!"
        random_list = {
			
			90 = { 
				compare_modifier = {
					value = scope:duel_value
					multiplier = 0.5
				}
				send_interface_toast = {
					title = "Killed by the might of the Voice"
					left_icon = scope:child
					scope:child = {
                        death = {
						    killer = root
						    death_reason = death_fight
					    }
                    }
				}
                send_interface_toast = {
					title =  "Killed by the might of the Voice"
					left_icon = scope:councillor
					scope:councillor = {
                        death = {
						    killer = root
						    death_reason = death_fight
					    }
                    }
				}
                send_interface_toast = {
					title =  "Killed by the might of the Voice"
					left_icon = scope:vassal
					scope:vassal = {
                        death = {
						    killer = root
						    death_reason = death_fight
					    }
                    }
				}
                add_trait = wounded_2
                add_character_modifier = unstable_thuum_modifier
			}
            
			10 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -0.5
				}
                send_interface_toast = {
					title = "Killed by the might of the Voice"
					left_icon = scope:child
					scope:child = {
                        death = {
						    killer = root
						    death_reason = death_fight
					    }
                    }
				}
                send_interface_toast = {
					title =  "Killed by the might of the Voice"
					left_icon = scope:councillor
					scope:councillor = {
                        death = {
						    killer = root
						    death_reason = death_fight
					    }
                    }
				}
                send_interface_toast = {
					title =  "Killed by the might of the Voice"
					left_icon = scope:vassal
					scope:vassal = {
                        death = {
						    killer = root
						    death_reason = death_fight
					    }
                    }
				}
				send_interface_toast = {
					title ="Killed by the might of the Voice"
					left_icon = scope:root
					death = {
						killer = root
						death_reason = death_fight
					}
				}
			}
		}
        add_character_flag = clavicus_done
    }
    
}
