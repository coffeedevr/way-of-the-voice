become_tongue_decision = {
	picture = { reference = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds" }
	decision_group_type = major
	ai_check_interval = 12

	desc = zero_stone_decision_desc
	selection_tooltip = zero_stone_decision_tooltip

	is_shown = {
		NOT = {
			has_character_flag = visited_gildergreen
		}
		has_character_flag = visit_gildergreen_as_decision
	}

	is_valid = {
		is_adult = yes
	}

	effect = {
		custom_tooltip = zero_stone_decision_effect_tt
		if = {
			limit = {
				capital_province.barony = title:b_gildergreen
			}
			trigger_event = ek_coming_of_age.0022
		}
		else = {
			start_travel_plan = {
				destination = title:b_gildergreen.title_province
				on_arrival_event = zzz_thuum_intro.0001
				on_arrival_destinations = all_but_last
			}
		}
	}

	ai_potential = {
		always = yes
	}
}
