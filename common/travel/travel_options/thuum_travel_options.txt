lok_option = {
	is_shown = {
   		has_trait = thuum_trait_track
	}

	is_valid = {
		has_perk = clear_skies_perk
		NOT = {
			has_perk = clear_skies_two_perk
			has_perk = clear_skies_three_perk
		}
	}
	
	travel_modifier = {
		travel_safety = 10
	}
	
	ai_will_do = {
		value = 100
	}
}
